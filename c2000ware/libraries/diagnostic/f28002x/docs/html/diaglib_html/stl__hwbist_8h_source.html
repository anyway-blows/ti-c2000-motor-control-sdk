292 static inline
293 void STL_HWBIST_injectError(const STL_HWBIST_Error errorType)
294 {
295  EALLOW;
296 
297  //
298  // Inject an error in the HWBIST.
299  //
300  HWREG(HWBIST_BASE + HWBIST_O_CSTCTEST) = (uint32_t)errorType;
301  NOP;
302  NOP;
303 
304  EDIS;
305 }
