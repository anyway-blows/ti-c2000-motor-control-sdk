91 //*****************************************************************************
92 //
93 // C-callable assembly routines to test the core registers for stuck at bits.
94 // They are called by the high level inline test functions below. They return
95 // STL_CPU_REG_PASS on success and STL_CPU_REG_FAIL on a failure.
96 //
97 // The scratchRAM parameter should be a pointer to RAM memory that will be
98 // used by the function to transfer data between the CPU and extended
99 // instruction set registers.
100 //
101 //*****************************************************************************
102 extern uint16_t STL_CPU_REG_testCPURegisters(bool injectError);
103 extern uint16_t STL_CPU_REG_testFPURegisters(bool injectError);
104 extern uint16_t STL_CPU_REG_testVCURegisters(uint32_t *scratchRAM,
105  bool injectError);

148 static inline uint16_t STL_CPU_REG_checkCPURegisters(bool injectError)
149 {

152  returnVal = STL_CPU_REG_testCPURegisters(injectError);
153 
154  //
155  // If test failed, set global error flag.
