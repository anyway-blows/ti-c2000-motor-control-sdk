135 static inline void ANGLE_GEN_run(ANGLE_GEN_Handle handle, const float32_t freq_Hz)
136 {
137  ANGLE_GEN_Obj *obj = (ANGLE_GEN_Obj *)handle;
138 
139  obj->freq_Hz = freq_Hz;
140  obj->angleDelta_rad = obj->freq_Hz * obj->angleDeltaFactor;
141 
142  float32_t angle_rad = obj->angle_rad;
143 
144  // increment the angle
145  angle_rad += obj->angleDelta_rad;
146 
147  //
148  // mask the angle for wrap around
149  // note: must account for the sign of the angle
150  //
151  if(angle_rad > MATH_PI)
152  {
153  angle_rad -= MATH_TWO_PI;
154  }
155  else if(angle_rad < -MATH_PI)
156  {
157  angle_rad += MATH_TWO_PI;
158  }
159 
160  obj->angle_rad = angle_rad;
161 
162  return;
163 } // end of ANGLE_GEN_run()
