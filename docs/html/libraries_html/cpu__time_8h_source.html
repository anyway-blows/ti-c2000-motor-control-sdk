140 CPU_TIME_updateCnts(CPU_TIME_Handle handle, const uint32_t cnt, uint16_t modIndex)
141 {
142  CPU_TIME_Obj *obj = (CPU_TIME_Obj *)handle;
143 
144  obj->timerCntPrev[modIndex] = cnt;
145 
146  if(obj->flag_resetStatus == true)
147  {

149 
150  uint16_t cnt;

153  {
154  obj->deltaMin[cnt] = 0xFFFF;

157  }
158 
159  return;

165 static inline void
166 CPU_TIME_run(CPU_TIME_Handle handle, uint32_t count, uint16_t modIndex)
167 {
168  CPU_TIME_Obj *obj = (CPU_TIME_Obj *)handle;
169 
170  obj->timerCntNow = count;
171 
172  if(modIndex >= MOD_INDEX_MAX)
173  {
174  return;
175  }
176 
177  obj->deltaNow[modIndex] =
178  (uint16_t)(obj->timerCntPrev[modIndex] - obj->timerCntNow);
179 
180  if(obj->deltaNow[modIndex] > obj->deltaMax[modIndex])
181  obj->deltaMax[modIndex] = obj->deltaNow[modIndex];
182 
183  if(obj->deltaNow[modIndex] < obj->deltaMin[modIndex])
184  obj->deltaMin[modIndex] = obj->deltaNow[modIndex];
185 
186  return;
187 } // end of CPU_TIME_run() function
