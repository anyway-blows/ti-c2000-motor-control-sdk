185 static inline void
186 DCLINK_SS_runCurrentReconstruction(DCLINK_SS_Handle handle,
187  const MATH_vec2 *pIdc1, const MATH_vec2 *pIdc2)
188 {
189  DCLINK_SS_Obj *obj = (DCLINK_SS_Obj *)handle;
190  MATH_vec2 Idc;
191 
192  if(obj->flagEnableFullSample)
193  {
194  if(obj->vecArea_1 == 0)
195  {
196  Idc.value[0] = (pIdc1->value[1] + pIdc2->value[0])*0.5f;
197  Idc.value[1] = (pIdc1->value[0] + pIdc2->value[1])*0.5f;
198  }
199  else // star area
200  {
201  if(obj->flag_SST_1 == 0) //measurement vector is on down count
202  {
203  Idc.value[0] = pIdc2->value[0];
204  Idc.value[1] = pIdc2->value[1];
205  }
206  else //measurement vector is on up count
207  {
208  Idc.value[0] = pIdc1->value[1];
209  Idc.value[1] = pIdc1->value[0];
210  }
211  }
212  }
213  else
214  {
215  if(obj->flag_SST_1 == 0) //measurement vector is on down count
216  {
217  Idc.value[0] = pIdc2->value[0];
218  Idc.value[1] = pIdc2->value[1];
219  }
220  else //measurement vector is on up count
221  {
222  Idc.value[0] = pIdc1->value[1];
223  Idc.value[1] = pIdc1->value[0];
224  }
225  }
226 
227  //
228  //Sector 1st_Sample 2nd_Sample Actual_Sector
229  // 1 1,1,0 (-Ic) 0,1,0 (+Ib) 2
230  // 2 1,0,1 (-Ib) 1,0,0 (+Ia) 6
231  // 3 1,1,0 (-Ic) 1,0,0 (+Ia) 1
232  // 4 0,1,1 (-Ia) 0,0,1 (+Ic) 4
233  // 5 0,1,1 (-Ia) 0,1,0 (+Ib) 3
234  // 6 1,0,1 (-Ib) 0,0,1 (+Ic) 5
235  //
236  switch(obj->sector_1)
237  {
238  case 1:
239  obj->I_A.value[2] = -Idc.value[0];
240  obj->I_A.value[1] = Idc.value[1];
241  obj->I_A.value[0] = -obj->I_A.value[1] - obj->I_A.value[2];
242  break;
243  case 2:
244  obj->I_A.value[1] = -Idc.value[0];
245  obj->I_A.value[0] = Idc.value[1];
246  obj->I_A.value[2] = -obj->I_A.value[0] - obj->I_A.value[1];
247  break;
248  case 3:
249  obj->I_A.value[2] = -Idc.value[0];
250  obj->I_A.value[0] = Idc.value[1];
251  obj->I_A.value[1] = -obj->I_A.value[0] - obj->I_A.value[2];
252  break;
253  case 4:
254  obj->I_A.value[0] = -Idc.value[0];
255  obj->I_A.value[2] = Idc.value[1];
256  obj->I_A.value[1] = -obj->I_A.value[0] - obj->I_A.value[2];
257  break;
258  case 5:
259  obj->I_A.value[0] = -Idc.value[0];
260  obj->I_A.value[1] = Idc.value[1];
261  obj->I_A.value[2] = -obj->I_A.value[0] - obj->I_A.value[1];
262  break;
263  case 6:
264  obj->I_A.value[1] = -Idc.value[0];
265  obj->I_A.value[2] = Idc.value[1];
266  obj->I_A.value[0] = -obj->I_A.value[1] - obj->I_A.value[2];
267  break;
268  default:
269  break;
270  }
271 
272  return;
273 } // end of DCLINK_SS_runCurrentReconstruction() function

291 static inline void
292 DCLINK_SS_runPWMCompensation(DCLINK_SS_Handle handle,
293  const MATH_vec2 *pVab_out, const float32_t Vdc_V,
294  MATH_ui_vec3 *pPwmCMPA, MATH_ui_vec3 *pPwmCMPB,
295  MATH_ui_vec2 *pUpSoc, MATH_ui_vec2 *pDownSoc)
296 {
297  DCLINK_SS_Obj *obj = (DCLINK_SS_Obj *)handle;
298  uint16_t cmp[3]; //cmp1,cmp2,cmp3;
299  uint16_t swap;
300  uint16_t sector = 0;
301  MATH_ui_vec3 newCMPA;
302  MATH_ui_vec3 newCMPB;
303  int16_t dT1,dT2;
304  int16_t half_dT1, half_dT2;
305  uint16_t pwmPRD = obj->pwmPeriod;
306  uint16_t index0 = 0;
307  uint16_t index1 = 1;
308  uint16_t index2 = 2;
309 
310  //
311  // determine the sector of space vector modulation
312  //
313  float32_t Va_tmp = pVab_out->value[1] * 0.5f;
314  float32_t Vb_tmp = MATH_SQRTTHREE_OVER_TWO * pVab_out->value[0];
315 
316  // Inverse clarke transformation
317  float32_t Va = pVab_out->value[1];
318  float32_t Vb = -Va_tmp + Vb_tmp;
319  float32_t Vc = -Va_tmp - Vb_tmp;
320 
321  if (Va > 0.) sector = 1;
322  if (Vb > 0.) sector = sector + 2;
323  if (Vc > 0.) sector = sector + 4;
324 
325  obj->sector_1 = obj->sector;
326  obj->sector = sector;
327 
328  //
329  // get EPWMx CMPA values
330  //
331  cmp[0] = pPwmCMPA->value[0];
332  cmp[1] = pPwmCMPA->value[1];
333  cmp[2] = pPwmCMPA->value[2];
334 
335  //
336  // find min, mid, max value of PWMx CMPA
337  // cmp[0]=min(max_duty), cmp[1]=mid, cmp[2]=max(min_duty)
338  //
339  if(cmp[0] > cmp[1])
340  {
341  swap = cmp[0];
342  cmp[0] = cmp[1];
343  cmp[1] = swap;
344 
345  index0 = 1;
346  index1 = 0;
347  }
348 
349  if(cmp[1] > cmp[2])
350  {
351  swap = cmp[1];
352  cmp[1] = cmp[2];
353  cmp[2] = swap;
354 
355  index2 = index1;
356  index1 = 2;
357  }
358 
359  if(cmp[0] > cmp[1])
360  {
361  swap = cmp[0];
362  cmp[0] = cmp[1];
363  cmp[1] = swap;
364 
365  swap = index1;
366  index1 = index0;
367  index0 = swap;
368  }
369 
370  //
371  // set default newCMPA/B value with new arranged compare value
372  //
373  newCMPA.value[0] = newCMPB.value[0] = cmp[0];
374  newCMPA.value[1] = newCMPB.value[1] = cmp[1];
375  newCMPA.value[2] = newCMPB.value[2] = cmp[2];
376 
377  obj->flag_SST_1 = obj->flag_SST;
378 
379  //
380  // sequence control for smooth injection at the sector transition
381  //
382  if(obj->flagEnableSequenceControl)
383  {
384 
385  float32_t Vs = sqrtf((pVab_out->value[0]*pVab_out->value[0]) +
386  (pVab_out->value[1]*pVab_out->value[1]));
387  float32_t highThr_V = obj->SSTOffThrVs_pu*Vdc_V;
388  float32_t lowThr_V = (obj->SSTOffThrVs_pu-0.05f)*Vdc_V;
389 
390  //
391  // check if sequence control should be disabled based on output modulation value,
392  // because ADC should be sampled at down count only in case of high duty
393  //
394  if(Vs > highThr_V)
395  {
396  // if the output voltage is higher than threshold, stop sequence control
397  obj->flagRunInHighModulation = true;
398  }
399  else if(Vs < lowThr_V)
400  {
401  // if the output voltage is higher than threshold, stop sequence control
402  obj->flagRunInHighModulation = false;
403  }
404 
405  if(obj->flagRunInHighModulation == false)
406  {
407  if((sector == 1) || (sector == 4) || (sector == 2)) //actual sector=2,4,6
408  {
409  // formal half = measurement vector, latter half = compensation vector
410  obj->flag_SST = 1;
411  }
412  else
413  {
414  // formal half = compensation vector, latter half = measurement vector
415  obj->flag_SST = 0;
416  }
417  }
418  else
419  {
420  // formal half = compensation vector, latter half = measurement vector
421  obj->flag_SST = 0;
422  }
423  }
424  else
425  {
426  // formal half = compensation vector, latter half = measurement vector
427  obj->flag_SST = 0;
428  }
429 
430  //
431  // PWM Phase Shift Compensation with Minimum Voltage Injection
432  //
433  dT1 = obj->minAvDuration - (cmp[2] - cmp[1]);
434  dT2 = obj->minAvDuration - (cmp[1] - cmp[0]);
435 
436  obj->vecArea_1 = obj->vecArea;
437 
438  if(obj->flag_SST == 0) // inject measurable vector at DOWN counter
439  {
440  if(dT1 > 0) // the 1st active vector is unmeasurable area
441  {
442  if(dT2 > 0) // the 2nd active vector is also unmeasurable area (low modulation)
443  {
444  obj->vecArea = 2;
445 
446  // cmp[2](min duty) is shifted to the left
447  newCMPA.value[2] = cmp[2] - dT1;
448  newCMPB.value[2] = cmp[2] + dT1;
449 
450  // cmp[0](max duty) is shifted to the right
451  newCMPA.value[0] = cmp[0] + dT2;
452  newCMPB.value[0] = cmp[0] - dT2;
453  }
454  else // in case of 6-vector(bar-area), only 1st active vector is unmeasurable
455  {
456  obj->vecArea = 1;
457 
458  half_dT1 = dT1>>1;
459 
460  // cmp[2](min duty) is shifted to the left
461  newCMPB.value[2] = __min(cmp[2] + half_dT1, pwmPRD);
462  newCMPA.value[2] = cmp[2] - half_dT1;
463 
464  // cmp[1](mid duty) is shifted to the right
465  newCMPB.value[1] = newCMPB.value[2] - obj->minAvDuration;
466  newCMPA.value[1] = __min(cmp[1]*2 - newCMPB.value[1], pwmPRD);
467 
468  int16_t new_dT2 = obj->minAvDuration - (newCMPB.value[1] - cmp[0]);
469  if(new_dT2 > 0)
470  {
471  // cmp[0](max duty) is shifted to the right
472  newCMPB.value[0] = cmp[0] - new_dT2;
473  newCMPA.value[0] = cmp[0] + new_dT2;
474  }
475  } // end of if(dT2 > 0)
476  } // end of if(dT1 > 0)
477  else // the 1st active vector is measurable
478  {
479  if(dT2 > 0) // only the 2nd active vector is unmeasurable area(bar-area)
480  {
481  obj->vecArea = 1;
482 
483  half_dT2 = dT2>>1;
484 
485  // cmp[0](max duty) is shifted to the right
486  newCMPB.value[0] = __max(cmp[0] - half_dT2, 0);
487  newCMPA.value[0] = cmp[0] + half_dT2;
488 
489  // cmp[1](mid duty) is shifted to the left
490  newCMPB.value[1] = newCMPB.value[0] + obj->minAvDuration;
491  newCMPA.value[1] = __max(cmp[1]*2 - newCMPB.value[1], 0);
492 
493  int16_t new_dT1 = obj->minAvDuration - (cmp[2] - newCMPB.value[1]);
494  if(new_dT1 > 0)
495  {
496  // cmp[2](min duty) is shifted to the right
497  newCMPB.value[2] = cmp[2] + new_dT1;
498  newCMPA.value[2] = cmp[2] - new_dT1;
499  }
500  }
501  else // all active vectors are measurable
502  {
503  obj->vecArea = 0;
504  }
505  }
506  }
507  else // inject measurable vector at UP counter (flag_SST = 1)
508  {
509  if(dT1 > 0) // the 2nd active vector is unmeasurable area
510  {
511  if(dT2 > 0) // the 1st active vector is also unmeasurable area (low modulation)
512  {
513  obj->vecArea = 2;
514 
515  // cmp[2](min duty) is shifted to the right
516  newCMPA.value[2] = cmp[2] + dT1;
517  newCMPB.value[2] = cmp[2] - dT1;
518 
519  // cmp[0](max duty) is shifted to the left
520  newCMPA.value[0] = cmp[0] - dT2;
521  newCMPB.value[0] = cmp[0] + dT2;
522  }
523  else // in case of 6-vector(bar-area), only 2nd active vector is unmeasurable
524  {
525  obj->vecArea = 1;
526 
527  half_dT1 = dT1>>1;
528 
529  // cmp[2](min duty) is shifted to the right
530  newCMPA.value[2] = __min(cmp[2] + half_dT1, pwmPRD);
531  newCMPB.value[2] = cmp[2] - half_dT1;
532 
533  // cmp[1](mid duty) is shifted to the left
534  newCMPA.value[1] = newCMPA.value[2] - obj->minAvDuration;
535  newCMPB.value[1] = __min(cmp[1]*2 - newCMPA.value[1], pwmPRD);
536 
537  int16_t new_dT2 = obj->minAvDuration - (newCMPA.value[1] - cmp[0]);
538  if(new_dT2 > 0)
539  {
540  // cmp[0](max duty) is shifted to the left
541  newCMPA.value[0] = cmp[0] - new_dT2;
542  newCMPB.value[0] = cmp[0] + new_dT2;
543  }
544  } // end of if(dT2 > 0)
545  } // end of if(dT1 > 0)
546  else // the 2nd active vector is measurable
547  {
548  if(dT2 > 0) // only the 1st active vector is unmeasurable area(bar-area)
549  {
550  obj->vecArea = 1;
551 
552  half_dT2 = dT2>>1;
553 
554  // cmp[0](max duty) is shifted to the left
555  newCMPA.value[0] = __max(cmp[0] - half_dT2, 0);
556  newCMPB.value[0] = cmp[0] + half_dT2;
557 
558  // cmp[1](mid duty) is shifted to the right
559  newCMPA.value[1] = newCMPA.value[0] + obj->minAvDuration;
560  newCMPB.value[1] = __max(cmp[1]*2 - newCMPA.value[1], 0);
561 
562  int16_t new_dT1 = obj->minAvDuration - (cmp[2] - newCMPA.value[1]);
563  if(new_dT1 > 0)
564  {
565  // cmp[2](min duty) is shifted to the right
566  newCMPA.value[2] = cmp[2] + new_dT1;
567  newCMPB.value[2] = cmp[2] - new_dT1;
568  }
569  } // end of if(dT2 > 0)
570  else // all active vectors are measurable
571  {
572  obj->vecArea = 0;
573  }
574  }
575  }
576 
577  //
578  // recalculate ADC SOC trigger points
579  //
580  pDownSoc->value[0] = __min((newCMPB.value[2] + newCMPB.value[1])>>1,
581  (newCMPB.value[2] - obj->sampleDelay));
582  pDownSoc->value[1] = __min((newCMPB.value[1] + newCMPB.value[0])>>1,
583  (newCMPB.value[1] - obj->sampleDelay));
584 
585  pUpSoc->value[0] = __max((newCMPA.value[0] + newCMPA.value[1])>>1,
586  (newCMPA.value[0] + obj->sampleDelay));
587  pUpSoc->value[1] = __max((newCMPA.value[2] + newCMPA.value[1])>>1,
588  (newCMPA.value[1] + obj->sampleDelay));
589 
590  //
591  // revise CMPA/B with new compare values
592  //
593  pPwmCMPA->value[index0] = newCMPA.value[0];
594  pPwmCMPB->value[index0] = newCMPB.value[0];
595  pPwmCMPA->value[index1] = newCMPA.value[1];
596  pPwmCMPB->value[index1] = newCMPB.value[1];
597  pPwmCMPA->value[index2] = newCMPA.value[2];
598  pPwmCMPB->value[index2] = newCMPB.value[2];
599 
600  return;
601 } // end of DCLINK_SS_runPWMCompensation() function
