85 #define DLOG_4CH_F_MACRO(v) \
86  switch(v.status) \
87  { \
88  case 1: /* wait for trigger*/ \
89  if(*v.input_ptr1>v.trig_value && v.prev_value<v.trig_value) \
90  { \
91  /* rising edge detected start logging data*/ \
92  v.status=2; \
93  } \
94  break; \
95  case 2: \
96  v.skip_count++; \
97  if(v.skip_count==v.pre_scalar) \
98  { \
99  v.skip_count=0; \
100  v.output_ptr1[v.count]=*v.input_ptr1; \
101  v.output_ptr2[v.count]=*v.input_ptr2; \
102  v.output_ptr3[v.count]=*v.input_ptr3; \
103  v.output_ptr4[v.count]=*v.input_ptr4; \
104  v.count++; \
105  if(v.count==v.size) \
106  { \
107  v.count=0; \
108  v.status=1; \
109  } \
110  } \
111  break; \
112  default: \
113  v.status=0; \
114  } \
115  v.prev_value=*v.input_ptr1; \
