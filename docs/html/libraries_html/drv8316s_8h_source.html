661 static inline DRV_Word_t DRV8316_buildCtrlWord(
662  const DRV8316_CtrlMode_e ctrlMode,
663  const DRV8316_Address_e regAddr,
664  const uint16_t data)
665 {
666  uint16_t p_addr = regAddr;
667  uint16_t p_data = data;
668  uint16_t p_mode = ctrlMode;
669 
670  uint16_t calc = (p_mode & 0x8000) | (p_addr & 0x7E00) | (p_data & 0x00FF);
671  uint16_t parity = 0;
672  while(calc)
673  {
674  parity ^= (calc & 1);
675  calc >>= 1;
676  }
677 
678  parity <<= 8;
