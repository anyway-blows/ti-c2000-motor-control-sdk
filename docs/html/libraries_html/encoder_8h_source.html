255 static inline uint16_t ENC_getHallState(ENC_Handle handle)
256 {
257  ENC_Obj *obj = (ENC_Obj *)handle;
258 
259  uint16_t hallState = 0;
260 
261  hallState = GPIO_readPin(obj->gpioHallU);
262  hallState += GPIO_readPin(obj->gpioHallV)<<1;
263  hallState += GPIO_readPin(obj->gpioHallW)<<2;
264 
265  return(hallState & 0x0007);

276 static inline void ENC_run(ENC_Handle handle)
277 {
278  ENC_Obj *obj = (ENC_Obj *)handle;
279 
280  if(obj->encState == ENC_CALIBRATION_DONE)
281  {
282  obj->thetaMech_pu = obj->mechanicalScaler *
283  (float32_t)EQEP_getPositionLatch(obj->qepHandle);
284 
285  obj->thetaMech_rad = obj->thetaMech_pu * MATH_TWO_PI;
286 
287  float32_t thetaElec_pu = obj->thetaMech_pu * obj->polePairs;
288 
289  obj->thetaElec_rad = (thetaElec_pu - ((int32_t)thetaElec_pu)) * MATH_TWO_PI;
290 
291  if(obj->thetaElec_rad >= MATH_PI)
292  {
293  obj->thetaElec_rad = obj->thetaElec_rad - MATH_TWO_PI;
294  }
295  else if(obj->thetaElec_rad <= -MATH_PI)
296  {
297  obj->thetaElec_rad = obj->thetaElec_rad + MATH_TWO_PI;
298  }
299  }
300  else if(obj->encState == ENC_WAIT_FOR_INDEX)
301  {
302  if(EQEP_getInterruptStatus(obj->qepHandle) & EQEP_INT_INDEX_EVNT_LATCH)
303  {
304  EQEP_setInitialPosition(obj->qepHandle, EQEP_getPositionLatch(obj->qepHandle));
305 
306  EQEP_setPositionInitMode(obj->qepHandle, EQEP_INIT_RISING_INDEX);
307 
308  obj->indexOffset = EQEP_getPositionLatch(obj->qepHandle);
309  obj->encState = ENC_CALIBRATION_DONE;
310  }
311  }
312  else // obj->encState == ENC_ALIGNMENT
313  {
314 
315 #if defined(ENC_CALIB)
316  obj->hallStateZero = ENC_getHallState(handle);
317  obj->hallIndex = 0;
318 #endif // ENC_CALIB
319 
320  // during alignment, reset the current shaft position to zero
321  EQEP_setPosition(obj->qepHandle, 0);
322 
323  // Reset pos cnt for QEP
324  EQEP_clearInterruptStatus(obj->qepHandle, EQEP_INT_INDEX_EVNT_LATCH);
325 
326  // reset poscnt init on index
327  EQEP_setPositionInitMode(obj->qepHandle, EQEP_INIT_DO_NOTHING);
328  }
329 
330 #if defined(ENC_CALIB)
331  obj->hallState = ENC_getHallState(handle);
332 
333  if(obj->hallState != obj->hallStatePrev)
334  {
335  obj->hallThetaData[obj->hallIndex] = obj->thetaElec_rad;
336  obj->hallPosData[obj->hallIndex] = EQEP_getPositionLatch(obj->qepHandle);
337  obj->hallStateIndex[obj->hallIndex] = obj->hallState;
338 
339  obj->hallIndex++;
340 
341  if(obj->hallIndex >= 12)
342  {
343  obj->hallIndex = 0;
344  }
345  }
346 
347  obj->hallStatePrev = obj->hallState;
348 #endif // ENC_CALIB
349 
350  return;
351 }
