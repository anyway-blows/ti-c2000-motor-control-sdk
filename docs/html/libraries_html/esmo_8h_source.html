399 static inline void ESMO_run(ESMO_Handle handle,
400  float32_t Vdcbus, MATH_vec3 *pVabc_pu, MATH_vec2 *pIabVec)
401 {
402  ESMO_Obj *obj = (ESMO_Obj *)handle;
403 
404  // Scale the incomming modulation functions with the DC bus voltage value
405  // and calculate the 3 Phase voltages
406  float32_t Vtemp = Vdcbus * obj->voltage_sf;
407 
408  float32_t VphaseA = Vtemp *
409  (pVabc_pu->value[0] * 2.0f - pVabc_pu->value[1] - pVabc_pu->value[2]);
410 
411  float32_t VphaseB = Vtemp *
412  (pVabc_pu->value[1] * 2.0f - pVabc_pu->value[0] - pVabc_pu->value[2]);
413 
414  // Voltage transformation (a,b,c) -> (Alpha,Beta)
415  obj->Valpha = VphaseA;
416  obj->Vbeta = (VphaseA + VphaseB * 2.0f) * MATH_ONE_OVER_SQRT_THREE;
417 
418  // Sliding mode current observer
419  float32_t ValphaError = obj->Valpha - obj->Ealpha - obj->Zalpha;
420  float32_t VbetaError = obj->Vbeta - obj->Ebeta - obj->Zbeta;
421 
422  obj->EstIalpha = obj->Gdsmopos * ValphaError + obj->Fdsmopos * obj->EstIalpha;
423  obj->EstIbeta = obj->Gqsmopos * VbetaError + obj->Fqsmopos * obj->EstIbeta;
424 
425  // Current errors
426  float32_t IalphaError = obj->EstIalpha - pIabVec->value[0] * obj->current_sf;
427  float32_t IbetaError = obj->EstIbeta - pIabVec->value[1] * obj->current_sf;
428 
429  // Sliding control calculator
430  obj->Zalpha = __fsat(IalphaError, obj->E0, -obj->E0) * obj->Kslide;
431  obj->Zbeta = __fsat(IbetaError, obj->E0, -obj->E0) * obj->Kslide;
432 
433  // Sliding control filter -> back EMF calculator
434  obj->Ealpha = obj->Ealpha + obj->Kslf * (obj->Zalpha - obj->Ealpha);
435  obj->Ebeta = obj->Ebeta + obj->Kslf * (obj->Zbeta - obj->Ebeta);
436 
437  // arc tangent of src radians
438  float32_t thetaOffset = __atan2puf32((obj->speedRef * obj->offsetSF), obj->Kslf);
439  obj->thetaPll = obj->theta - thetaOffset;
440 
441  float32_t pllSine = __sinpuf32(obj->thetaPll);
442  float32_t pllCosine = __cospuf32(obj->thetaPll);
443 
444  obj->Ed = obj->Ealpha * pllCosine + obj->Ebeta * pllSine;
445  obj->Eq = obj->Ebeta * pllCosine - obj->Ealpha * pllSine;
446  obj->Eq_mag = sqrtf(obj->Ealpha * obj->Ealpha + obj->Ebeta * obj->Ebeta);
447 
448  //0.1591549431 = 1/6.28 (1/(2*PI())
449  float32_t thetaErrSF = obj->thetaErrSF;
450 
451  if(obj->Eq >= 0.0f)
452  {
453  thetaErrSF = -obj->thetaErrSF;
454  }
455 
456  obj->thetaErr = obj->Ed * thetaErrSF / obj->Eq_mag;
457 
458  // integral term
459  obj->pll_ui = (obj->pll_Ki * obj->thetaErr) + obj->pll_ui;
460 
461  // control output
462  obj->pll_Out = __fsat((obj->pll_Kp * obj->thetaErr + obj->pll_ui),
463  obj->pll_Umax, obj->pll_Umin);
464 
465  obj->speedEst = (obj->pll_Out + obj->speedEst) * 0.5f;
466 
467  // low pass filter for estimation speed
468  obj->speedFlt = obj->lpf_b0 * obj->pll_Out + obj->lpf_a1 * obj->speedFlt;
469 
470  // speed integrate
471  obj->theta = obj->theta + obj->speedFlt * obj->thetaDelta;
472 
473  if(obj->theta > 1.0f)
474  {
475  obj->theta -= 1.0f;
476  }
477  else if(obj->theta < -1.0f)
478  {
479  obj->theta += 1.0f;
480  }
481 
482  obj->thetaEst = obj->theta * MATH_TWO_PI;
483 
484  if(obj->thetaEst > MATH_PI)
485  {
486  obj->thetaEst -= MATH_TWO_PI;
487  }
488  else if(obj->thetaEst < (-MATH_PI))
489  {
490  obj->thetaEst += MATH_TWO_PI;
491  }
492 
493  return;
494 } // end of ESMO_run() function
