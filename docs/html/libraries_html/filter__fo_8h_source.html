259 static inline float32_t
260 FILTER_FO_run(FILTER_FO_Handle handle, const float32_t inputValue)
261 {
262  FILTER_FO_Obj *obj = (FILTER_FO_Obj *)handle;
263 
264  float32_t a1 = obj->a1;
265  float32_t b0 = obj->b0;
266  float32_t b1 = obj->b1;
267  float32_t x1 = obj->x1;
268  float32_t y1 = obj->y1;
269  float32_t y0 = 1.0;
270  float32_t x0 = inputValue;
271 
272  //
273  // Compute the output
274  //
275  y0 = (b0 * x0) + (b1 * x1) - (a1 * y1);
276  // y0 = (b0 * inputValue) + (a1 * y1);
277 
278  //
279  // Store values for next time
280  //
281  obj->x1 = inputValue;
282  obj->y1 = y0;
283 
284  return(y0);
285 } // end of FILTER_FO_run() function
