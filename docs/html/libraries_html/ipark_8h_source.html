184 static inline void
185 IPARK_run(IPARK_Handle handle, const MATH_Vec2 *pInVec, MATH_Vec2 *pOutVec)
186 {
187  IPARK_Obj *obj = (IPARK_Obj *)handle;
188 
189  float32_t sinTh = obj->sinTh;
190  float32_t cosTh = obj->cosTh;
191 
192  float32_t value_0 = pInVec->value[0];
193  float32_t value_1 = pInVec->value[1];
194 
195  pOutVec->value[0] = (value_0 * cosTh) - (value_1 * sinTh);
196  pOutVec->value[1] = (value_1 * cosTh) + (value_0 * sinTh);
197 
198  return;
199 } // end of IPARK_run() function
