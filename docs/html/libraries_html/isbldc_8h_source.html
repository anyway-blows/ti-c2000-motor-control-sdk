270 static inline void ISBLDC_run(ISBLDC_Handle handle, MATH_Vec3 *pVabcVec)
271 {
272  ISBLDC_Obj *obj = (ISBLDC_Obj *)handle;
273 
274  // State Filter decides integration direction and bemf phase to sense
275  switch(obj->commState)
276  {
277  case 0:
278  obj->bemfPhase = PHS_C;
279  obj->bemfDirect = DIR_NEG;
280  break;
281 
282  case 1:
283  obj->bemfPhase = PHS_B;
284  obj->bemfDirect = DIR_POS;
285  break;
286 
287  case 2:
288  obj->bemfPhase = PHS_A;
289  obj->bemfDirect = DIR_NEG;
290  break;
291 
292  case 3:
293  obj->bemfPhase = PHS_C;
294  obj->bemfDirect = DIR_POS;
295  break;
296 
297  case 4:
298  obj->bemfPhase = PHS_B;
299  obj->bemfDirect = DIR_NEG;
300  break;
301 
302  case 5:
303  obj->bemfPhase = PHS_A;
304  obj->bemfDirect = DIR_POS;
305  }
306 
307  obj->Vabcg.value[0] = pVabcVec->value[0] - obj->VabcOffset.value[0];
308  obj->Vabcg.value[1] = pVabcVec->value[1] - obj->VabcOffset.value[1];
309  obj->Vabcg.value[2] = pVabcVec->value[2] - obj->VabcOffset.value[2];
310 
311  obj->Vabcn.value[0] = obj->Vabcg.value[0] -
312  ((obj->Vabcg.value[1] + obj->Vabcg.value[2]) * 0.5f);
313  obj->Vabcn.value[1] = obj->Vabcg.value[1] -
314  ((obj->Vabcg.value[0] + obj->Vabcg.value[2]) * 0.5f);
315  obj->Vabcn.value[2] = obj->Vabcg.value[2] -
316  ((obj->Vabcg.value[0] + obj->Vabcg.value[1]) * 0.5f);
317 
318  switch(obj->bemfPhase)
319  {
320  case PHS_A:
321  // Point to phase voltage A to neutral
322  obj->VintPhase = obj->Vabcn.value[0];
323  break;
324 
325  case PHS_B:
326  // Point to phase voltage B to neutral
327  obj->VintPhase = obj->Vabcn.value[1];
328  break;
329 
330  case PHS_C:
331  // Point to phase voltage C to neutral
332  obj->VintPhase = obj->Vabcn.value[2];
333  break;
334  }
335 
336  obj->commTrigFlag = false;
337 
338  if (obj->bemfDirect == DIR_POS)
339  {
340  // Integrate positive
341  if(obj->bemfLockFlag == true)
342  {
343  if(obj->VintPhase < 0.0f)
344  {
345  obj->bemfLockFlag = false;
346  }
347  }
348  else
349  {
350  obj->bemfInt += obj->VintPhase; // Integrate the phase voltage
351 
352  if(obj->bemfInt > 0.0f) // Only integrate positively
353  {
354  // Check to see if there is enough volt-seconds to commutate
355  if(obj->bemfInt > obj->thresholdInt)
356  {
357  obj->bemfLockFlag = true;
358  obj->commTrigFlag = true;
359  obj->bemfInt = 0.0f;
360 
361  //#####BEGIN_INTERNAL#####
362  #ifdef ISBLDC_DEBUG
363  GPIO_togglePin(GPIO_ZCP_TRIG);
364  #endif // ISBLDC_DEBUG
365  //#####END_INTERNAL#######
366  }
367  }
368  else
369  {
370  obj->bemfInt = 0.0f;
371  }
372  }
373  }
374  else
375  {
376  // Integrate negative
377  if(obj->bemfLockFlag == true)
378  {
379  if(obj->VintPhase > 0.0f)
380  {
381  obj->bemfLockFlag = false;
382  }
383  }
384  else
385  {
386  obj->bemfInt += obj->VintPhase; // Integrate the phase voltage
387 
388  if(obj->bemfInt < 0.0f) // Only integrate negatively
389  {
390  // Check to see if there is enough volt-seconds to commutate
391  if(obj->bemfInt < -obj->thresholdInt)
392  {
393  obj->bemfLockFlag = true;
394  obj->commTrigFlag = true;
395  obj->bemfInt = 0.0f;
396 
397  //#####BEGIN_INTERNAL#####
398  // only for debug
399  #ifdef ISBLDC_DEBUG
400  GPIO_togglePin(GPIO_ZCP_TRIG);
401  #endif // ISBLDC_DEBUG
402  //#####END_INTERNAL#######
403  }
404  }
405  else
406  {
407  obj->bemfInt = 0.0f;
408  }
409  }
410  }
411 
412  // calculate speed
413  obj->timeStamp++;
414 
415  if(obj->commTrigFlag == true)
416  {
417  //#####BEGIN_INTERNAL#####
418  //only for debug
419  #ifdef ISBLDC_DEBUG
420  GPIO_togglePin(GPIO_PHS_COMM);
421  #endif // ISBLDC_DEBUG
422  //#####END_INTERNAL#######
423 
424  obj->timePeriod = (obj->timeStamp + obj->timeStampBuf[0] +
425  obj->timeStampBuf[1] + obj->timeStampBuf[2] +
426  obj->timeStampBuf[3] + obj->timeStampBuf[4] );
427 
428  obj->timeStampBuf[4] = obj->timeStampBuf[3];
429  obj->timeStampBuf[3] = obj->timeStampBuf[2];
430  obj->timeStampBuf[2] = obj->timeStampBuf[1];
431  obj->timeStampBuf[1] = obj->timeStampBuf[0];
432  obj->timeStampBuf[0] = obj->timeStamp;
433 
434  obj->speedInt_Hz = obj->speedScaler / ((float32_t)obj->timePeriod);
435 
436  obj->timeStamp = 0;
437  obj->intTimer = 0;
438  }
439 
440  obj->intTimer++;
441 
442  if(obj->intTimer > 20000)
443  {
444  obj->intTimer = 0;
445  obj->speedInt_Hz = 0.0f;
446  }
447 
448  return;
449 }
