164 static inline void
165 MOD6CNT_run(MOD6CNT_Handle handle, const float32_t speedRef_Hz)
166 {
167  MOD6CNT_Obj *obj = (MOD6CNT_Obj *)handle;
168 
169  if(obj->flagCount == true)
170  {
171  if(speedRef_Hz >= 0.0f)
172  {
173  if(obj->counter == obj->counterMax)
174  {
175  obj->counter = 0;
176  }
177  else
178  {
179  obj->counter++;
180  }
181  }
182  else
183  {
184  if(obj->counter == 0)
185  {
186  obj->counter = 5;
187  }
188  else
189  {
190  obj->counter--;
191  }
192  }
193  }
194 
195  return;
196 } // end of MOD6CNT_run()
