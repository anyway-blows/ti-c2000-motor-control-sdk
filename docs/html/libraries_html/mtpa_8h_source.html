336 static inline void
337 MTPA_computeCurrentReference(MTPA_Handle handle, const float32_t Is_ref_A)
338 {
339  MTPA_Obj *obj = (MTPA_Obj *)handle;
340 
341 
342  obj->Is_ref_A = MATH_abs(Is_ref_A);
343 
344  if(MTPA_getFlagEnable(handle) == true)
345  {
346  // Perform the Maximum torque per ampere (MTPA)
347  if((obj->kconst != 0.0) && (obj->Is_ref_A != 0.0))
348  {
349 
350  #ifdef __TMS320C28XX_CLA__
351  obj->Idq_ref_A.value[0] = obj->kconst -
352  CLAsqrt((obj->kconst * obj->kconst) -
353  (0.5 * (obj->Is_ref_A * obj->Is_ref_A)));
354 
355  obj->Idq_ref_A.value[1] = CLAsqrt((obj->Is_ref_A * obj->Is_ref_A) -
356  (obj->Idq_ref_A.value[0] * obj->Idq_ref_A.value[0]));
357  #else
358  obj->Idq_ref_A.value[0] = obj->kconst -
359  (float32_t)sqrt((obj->kconst * obj->kconst) +
360  (0.5 * (obj->Is_ref_A * obj->Is_ref_A)));
361 
362  obj->Idq_ref_A.value[1] = sqrt((obj->Is_ref_A * obj->Is_ref_A) -
363  (obj->Idq_ref_A.value[0] * obj->Idq_ref_A.value[0]));
364  #endif
365 
366  if(Is_ref_A < 0.0)
367  {
368  obj->Idq_ref_A.value[1] = -obj->Idq_ref_A.value[1];
369  }
370  }
371  else
372  {
373  obj->Idq_ref_A.value[0] = 0.0;
374  obj->Idq_ref_A.value[1] = Is_ref_A;
375  }
376  }
377  else
378  {
379  obj->Idq_ref_A.value[0] = 0.0;
380  obj->Idq_ref_A.value[1] = Is_ref_A;
381  }
382 
383  return;
384 } // end of MTPA_computeCurrentReference() function

393 static inline float32_t
394 MTPA_computeCurrentAngle(MTPA_Handle handle, const float32_t Is_ref_A)
395 {
396  MTPA_Obj *obj = (MTPA_Obj *)handle;
397 
398  obj->Is_ref_A = fabsf(Is_ref_A);
399 
400  if(MTPA_getFlagEnable(handle) == true)
401  {
402  // Perform the Maximum torque per ampere (MTPA)
403  if((obj->kconst != 0.0) && (obj->Is_ref_A != 0.0))
404  {
405  obj->gconst = obj->kconst / obj->Is_ref_A;
406 
407  #ifdef __TMS320C28XX_CLA__
408  obj->angleCurrent_rad = CLAacos(obj->gconst -
409  CLAsqrt(obj->gconst * obj->gconst + 0.5));
410  #else
411  obj->angleCurrent_rad = acosf(obj->gconst -
412  sqrtf(obj->gconst * obj->gconst + 0.5));
413  #endif
414 
415  }
416  else
417  {
418  obj->angleCurrent_rad = MATH_PI_OVER_TWO;
419  }
420  }
421  else
422  {
423  obj->angleCurrent_rad = MATH_PI_OVER_TWO;
424  }
425 
426  return(obj->angleCurrent_rad);
427 } // end of MTPA_computeCurrentAngle() function
