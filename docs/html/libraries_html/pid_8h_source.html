632 static inline void
633 PID_run_parallel(PID_Handle handle, const float32_t refValue,
634  const float32_t fbackValue, const float32_t ffwdValue,
635  float32_t *pOutValue)
636 {
637  PID_Obj *obj = (PID_Obj *)handle;
638 
639  float32_t Error;
640  float32_t Kp = PID_getKp(handle);
641  float32_t Ki = PID_getKi(handle);
642  float32_t Kd = PID_getKd(handle);
643  float32_t Up;
644  float32_t Ui = PID_getUi(handle);
645  float32_t Ud_tmp,Ud;
646  float32_t outMax = PID_getOutMax(handle);
647  float32_t outMin = PID_getOutMin(handle);
648 
649  Error = refValue - fbackValue;
650 
651  //
652  // Compute the proportional output
653  //
654  Up = Kp * Error;
655 
656  //
657  // Compute the integral output
658  //
659  Ui = MATH_sat(Ui + (Ki * Error),outMax,outMin);
660 
661  Ud_tmp = Kd * Error;
662  Ud = FILTER_FO_run(obj->derFilterHandle,Ud_tmp);
663 
664  //
665  // Compute the derivative term
666  //
667  PID_setUi(handle,Ui);
668  PID_setRefValue(handle,refValue);
669  PID_setFbackValue(handle,fbackValue);
670  PID_setFfwdValue(handle,ffwdValue);
671 
672  //
673  // Saturate the output
674  //
675  *pOutValue = MATH_sat(Up + Ui + Ud + ffwdValue,outMax,outMin);
676 
677  return;
678 } // end of PID_run_parallel() function
