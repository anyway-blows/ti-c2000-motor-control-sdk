62 //
63 // Used in translating the absolute encoder position change
64 // an equivalent incremental position change.
65 //
66 // PTO_ABS2QEP_SAMPLE_ADJUST takes into account time required to
67 // load the next PTO sample. If this adjustment is not used then
68 // the PTO will be slightly longer than the sampling period. This
69 // time will accumulate over samples if no adjustment is made.
70 //
71 #define PTO_ABS2QEP_ABS_MAX_POSITION \
72  (1UL << PTO_ABS2QEP_ABS_ENCODER_RESOLUTION)
