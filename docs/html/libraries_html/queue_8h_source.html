247 static inline status
248 QUEUE_postEventFirst(QUEUE_Handle handle, const EVENT_Fxn eventFxn,
249  const EVENT_ArgList *pArgList,
250  const uint_least8_t numArgs)
251 {
252  if(numArgs > EVENT_MAX_NUM_ARGS)
253  {
254  return(ERROR);
255  }
256  else
257  {
258  QUEUE_Obj *obj = (QUEUE_Obj *)handle;
259  EVENT_Obj *pEvent;
260  EVENT_Obj *pFirstEvent;
261 
262  pEvent = &gEvents[gEventIndex];
263 
264  if(pEvent->taken)
265  {
266  return(QUEUE_STATUS_FIRST_EVENT_TAKEN);
267  }
268 
269  gEventIndex++;
270  gEventIndex &= QUEUE_MAX_NUM_EVENTS - 1;
271 
272  pEvent->taken = true;
273 
274  pEvent->eventFxn = eventFxn;
275  pEvent->argList.arg[0] = pArgList->arg[0];
276  pEvent->argList.arg[1] = pArgList->arg[1];
277  pEvent->argList.arg[2] = pArgList->arg[2];
278  pEvent->argList.arg[3] = pArgList->arg[3];
279 
280  //
281  // Set the number of arguments
282  //
283  pEvent->numArgs = numArgs;
284 
285  //
286  // Insert the event at top of the queue
287  //
288  pFirstEvent = (EVENT_Obj *)obj->firstEvent;
289 
290  pEvent->prevEvent = pFirstEvent->prevEvent;
291  pEvent->nextEvent = pFirstEvent;
292  pFirstEvent->prevEvent = pEvent;
293  obj->firstEvent = (EVENT_Handle)pEvent;
294 
295  return(OK);
296  }
297 } // end of QUEUE_postEventFirst() function

459 static inline void
460 QUEUE_executeEvent(QUEUE_Handle handle)
461 {
462  QUEUE_Obj *obj = (QUEUE_Obj *)handle;
463  EVENT_Obj *pEvent = (EVENT_Obj *)obj->firstEvent;
464  uint_least8_t numArgs = pEvent->numArgs;
465 
466  //
467  // Based on the number of event arguments, make the appropriate event call
468  //
469  if(numArgs == 4)
470  {
471  (pEvent->eventFxn)(pEvent->argList.arg[0],pEvent->argList.arg[1],
472  pEvent->argList.arg[2],pEvent->argList.arg[3]);
473  }
474  else if(numArgs == 3)
475  {
476  (pEvent->eventFxn)(pEvent->argList.arg[0],pEvent->argList.arg[1],
477  pEvent->argList.arg[2]);
478  }
479  else if(numArgs == 2)
480  {
481  (pEvent->eventFxn)(pEvent->argList.arg[0],pEvent->argList.arg[1]);
482  }
483  else if(numArgs == 1)
484  {
485  (pEvent->eventFxn)(pEvent->argList.arg[0]);
486  }
487  else if(numArgs == 0)
488  {
489  (pEvent->eventFxn)();
490  }
491 
492  //
493  // Remove the event from the queue
494  //
495  QUEUE_removeFirstEvent(handle);
496 
497  return;
498 } // end of QUEUE_executeEvent() function
