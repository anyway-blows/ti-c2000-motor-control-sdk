165 static inline void RIMPULSE_run(RIMPULSE_Handle handle)
166 {
167  RIMPULSE_Obj *obj = (RIMPULSE_Obj *)handle;
168 
169  if(obj->periodOut <= obj->desiredOutEnd)
170  {
171  obj->ramp3DoneFlag = true;
172  }
173  else
174  {
175  obj->ramp3DelayCount++;
176 
177  if(obj->ramp3DelayCount >= obj->ramp3Delay)
178  {
179  obj->periodOut--;
180 
181  if(obj->periodOut < obj->desiredOutMin)
182  {
183  obj->periodOut = obj->desiredOutMin;
184  }
185 
186  obj->ramp3DelayCount = 0;
187  }
188  }
189 
190  obj->trigFlag = false;
191 
192  obj->counter++;
193 
194  if(obj->counter >= obj->periodOut)
195  {
196  obj->trigFlag = true;
197  obj->counter = 0;
198  }
199 
200  return;
201 } // end of RIMPULSE_run()
