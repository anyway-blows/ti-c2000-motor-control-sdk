116 static inline void SPDFR_run(SPDFR_Handle handle, float32_t theta)
117 {
118  SPDFR_Obj *obj = (SPDFR_Obj *)handle;
119  float32_t error;
120 
121  obj->theta = theta * MATH_ONE_OVER_TWO_PI;
122 
123  // error cal
124  error = obj->theta - obj->thetaPrev;
125 
126  if(error < -0.5f)
127  {
128  error += 1.0f;
129  }
130  else if(error > 0.5f)
131  {
132  error -= 1.0f;
133  }
134 
135  // Update the electrical angle
136  obj->thetaPrev = obj->theta;
137 
138  // Low-pass filter
139  obj->speed_pu = (obj->K2 * obj->speed_pu) + (obj->K3 * error);
140 
141  // Saturate the output
142  obj->speed_pu = __fsat(obj->speed_pu, 1.0f, -1.0f);
143 
144  // Change motor speed for pu to rpm value
145  obj->speed_Hz = obj->scaleFreq * obj->speed_pu;
146 
147  return;
148 }
