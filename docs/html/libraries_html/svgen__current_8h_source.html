281 SVGENCURRENT_RunIgnoreShunt(SVGENCURRENT_Handle handle,
282  uint16_t cmp1, uint16_t cmp2, uint16_t cmp3,

289 
290  uint16_t nextPulse1 = (cmp1 + cmpM1)>>1;

292  uint16_t nextPulse3 = (cmp3 + cmpM3)>>1;
293

306  obj->ignoreShunt = SVGENCURRENT_IGNORE_A;
307  }
308  }
309  else if(nextPulse2 < minWidth)

311  if((nextPulse1 < minWidth) || ((cmp1 - cmp2) < minWidth))

313  obj->ignoreShunt = SVGENCURRENT_IGNORE_AB;
314  }

352 static inline void
353 SVGENCURRENT_RunRegenCurrent(SVGENCURRENT_Handle handle,
354  MATH_Vec3 *pADCData, MATH_Vec3 *pADCDataPrev)
355 {
356  SVGENCURRENT_Obj *obj = (SVGENCURRENT_Obj *)handle;
357 
358  float32_t Ia = pADCData->value[0];
359  float32_t Ib = pADCData->value[1];
360  float32_t Ic = pADCData->value[2];
361 
362  // select valid shunts and ignore one when needed
363  if(obj->ignoreShunt == SVGENCURRENT_IGNORE_A)
364  {
365  // repair a based on b and c
366  Ia = -Ib - Ic; //Ia = -Ib - Ic;
367  }
368  else if(obj->ignoreShunt == SVGENCURRENT_IGNORE_B)
369  {
370  // repair b based on a and c
371  Ib = -Ia - Ic; //Ib = -Ia - Ic;
372  }
373  else if(obj->ignoreShunt == SVGENCURRENT_IGNORE_C)
374  {
375  // repair c based on a and b
376  Ic = -Ia - Ib; //Ic = -Ia - Ib;
377  }
378  else if(obj->ignoreShunt == SVGENCURRENT_IGNORE_AB)
379  {
380  Ia = -Ic * 0.5f;
381  Ib = Ia;
382  }
383  else if(obj->ignoreShunt == SVGENCURRENT_IGNORE_AC)
384  {
385  Ia = -Ib * 0.5f;
386  Ic = Ia;
387  }
388  else if(obj->ignoreShunt == SVGENCURRENT_IGNORE_BC)
389  {
390  Ib = -Ia * 0.5f;
391  Ic = Ib;
392  }
393 
394  pADCData->value[0] = Ia;
395  pADCData->value[1] = Ib;
396  pADCData->value[2] = Ic;
397 
398  pADCDataPrev->value[0] += (pADCData->value[0] - pADCDataPrev->value[0]) * 0.5f;
399  pADCDataPrev->value[1] += (pADCData->value[1] - pADCDataPrev->value[1]) * 0.5f;
400  pADCDataPrev->value[2] += (pADCData->value[2] - pADCDataPrev->value[2]) * 0.5f;
401 
402  if(obj->compMode > SVGENCURRENT_TWO_PHASE_MEASURABLE)
403  {
404  pADCData->value[0] = pADCDataPrev->value[0];
405  pADCData->value[1] = pADCDataPrev->value[1];
406  pADCData->value[2] = pADCDataPrev->value[2];
407  }
408 
409  return;
410 } // end of SVGENCURRENT_RunRegenCurrent() function

417 static inline void
418 SVGENCURRENT_compPWMData(SVGENCURRENT_Handle handle,
419  MATH_Vec3 *pPWMData, MATH_Vec3 *pPWMData_prev)
420 {
421  SVGENCURRENT_Obj *obj = (SVGENCURRENT_Obj *)handle;
422 
423  float32_t Va_avg = (pPWMData->value[0] + pPWMData_prev->value[0]) * 0.5f;
424  float32_t Vb_avg = (pPWMData->value[1] + pPWMData_prev->value[1]) * 0.5f;
425  float32_t Vc_avg = (pPWMData->value[2] + pPWMData_prev->value[2]) * 0.5f;
426 
427  float32_t Vlimit = obj->Vlimit;
428  float32_t Vmid, Vmid_prev, Voffset;
429 
430  //define compensation mode
431  if(Va_avg > Vlimit)
432  {
433  if(Vb_avg > Vlimit)
434  {
435  obj->compMode = SVGENCURRENT_ONE_PHASE_MEASURABLE;
436 
437  if(Va_avg > Vb_avg)
438  {
439  obj->Vmid = SVGENCURRENT_VMID_B;
440  Vmid = pPWMData->value[1];
441  Vmid_prev = pPWMData_prev->value[1];
442  }
443  else
444  {
445  obj->Vmid = SVGENCURRENT_VMID_A;
446  Vmid = pPWMData->value[0];
447  Vmid_prev = pPWMData_prev->value[0];
448  }
449  }
450  else if(Vc_avg > Vlimit)
451  {
452  obj->compMode = SVGENCURRENT_ONE_PHASE_MEASURABLE;
453 
454  if(Va_avg > Vc_avg)
455  {
456  obj->Vmid = SVGENCURRENT_VMID_C;
457  Vmid = pPWMData->value[2];
458  Vmid_prev = pPWMData_prev->value[2];
459  }
460  else
461  {
462  obj->Vmid = SVGENCURRENT_VMID_A;
463  Vmid = pPWMData->value[0];
464  Vmid_prev = pPWMData_prev->value[0];
465  }
466  }
467  else
468  {
469  obj->compMode = SVGENCURRENT_TWO_PHASE_MEASURABLE;
470 
471  if(Vb_avg > Vc_avg)
472  {
473  obj->Vmid = SVGENCURRENT_VMID_B;
474  }
475  else
476  {
477  obj->Vmid = SVGENCURRENT_VMID_C;
478  }
479  }
480  }
481  else
482  {
483  if(Vb_avg > Vlimit)
484  {
485  if(Vc_avg > Vlimit)
486  {
487  obj->compMode = SVGENCURRENT_ONE_PHASE_MEASURABLE;
488 
489  if(Vb_avg > Vc_avg)
490  {
491  obj->Vmid = SVGENCURRENT_VMID_C;
492  Vmid = pPWMData->value[2];
493  Vmid_prev = pPWMData_prev->value[2];
494  }
495  else
496  {
497  obj->Vmid = SVGENCURRENT_VMID_B;
498  Vmid = pPWMData->value[1];
499  Vmid_prev = pPWMData_prev->value[1];
500  }
501  }
502  else
503  {
504  obj->compMode = SVGENCURRENT_TWO_PHASE_MEASURABLE;
505 
506  if(Va_avg > Vc_avg)
507  {
508  obj->Vmid = SVGENCURRENT_VMID_A;
509  }
510  else
511  {
512  obj->Vmid = SVGENCURRENT_VMID_C;
513  }
514  }
515  }
516  else if(Vc_avg > Vlimit)
517  {
518  obj->compMode = SVGENCURRENT_TWO_PHASE_MEASURABLE;
519 
520  if(Va_avg > Vb_avg)
521  {
522  obj->Vmid = SVGENCURRENT_VMID_A;
523  }
524  else
525  {
526  obj->Vmid = SVGENCURRENT_VMID_B;
527  }
528  }
529  else
530  {
531  obj->compMode = SVGENCURRENT_ALL_PHASE_MEASURABLE;
532  obj->ignoreShunt = SVGENCURRENT_USE_ALL;
533  }
534  }
535 
536  //phase voltage compensator
537  if(obj->compMode > SVGENCURRENT_TWO_PHASE_MEASURABLE)
538  {
539  Voffset = (Vmid + Vmid_prev) * 0.5f - Vlimit;
540 
541  if(pPWMData->value[0] > -0.50f)
542  {
543  pPWMData->value[0] -= Voffset;
544  }
545 
546  if(pPWMData->value[1] > -0.50f)
547  {
548  pPWMData->value[1] -= Voffset;
549  }
550 
551  if(pPWMData->value[2] > -0.50f)
552  {
553  pPWMData->value[2] -= Voffset;
554  }
555 
556  obj->Voffset = Voffset;
557  }
558 
559  // get ignore current
560  if(((pPWMData->value[0] + pPWMData_prev->value[0]) * 0.5f) > Vlimit)
561  {
562  obj->ignoreShunt = SVGENCURRENT_IGNORE_A;
563  }
564  else if(((pPWMData->value[1] + pPWMData_prev->value[1]) * 0.5f) > Vlimit)
565  {
566  obj->ignoreShunt = SVGENCURRENT_IGNORE_B;
567  }
568  else if(((pPWMData->value[2] + pPWMData_prev->value[2]) * 0.5f) > Vlimit)
569  {
570  obj->ignoreShunt = SVGENCURRENT_IGNORE_C;
571  }
572  else
573  {
574  obj->ignoreShunt = SVGENCURRENT_USE_ALL;
575  }
576 
577 
578  pPWMData_prev->value[0] = pPWMData->value[0];
579  pPWMData_prev->value[1] = pPWMData->value[1];
580  pPWMData_prev->value[2] = pPWMData->value[2];
581 
582  return;
583 } // end of SVGENCURRENT_compPWMData() function
