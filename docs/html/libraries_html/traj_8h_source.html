307 static inline void
308 TRAJ_run(TRAJ_Handle handle)
309 {
310  float32_t targetValue = TRAJ_getTargetValue(handle);
311  float32_t intValue = TRAJ_getIntValue(handle);
312  float32_t error = targetValue - intValue;
313  float32_t maxDelta = TRAJ_getMaxDelta(handle);
314  float32_t minValue = TRAJ_getMinValue(handle);
315  float32_t maxValue = TRAJ_getMaxValue(handle);
316 
317 
318 #ifdef __TMS320C28XX_CLA__
319  //
320  // Increment the value
321  //
322  intValue += MATH_sat(error, maxDelta, -maxDelta);
323 
324  //
325  // Bound the value
326  //
327  intValue = MATH_sat(intValue, maxValue, minValue);
328 #else
329  //
330  // Increment the value
331  //
332  intValue += __fsat(error, maxDelta, -maxDelta);
333 
334  //
335  // Bound the value
336  //
337  intValue = __fsat(intValue, maxValue, minValue);
338 #endif // __TMS320C28XX_CLA__
339 
340  //
341  // Store the value
342  //
343  TRAJ_setIntValue(handle,intValue);
344 
345  return;
346 } // end of TRAJ_run() function
