236 inline void VSF_setPeriod(VSF_Handle vsfHandle)
237 {
238  VSF_Obj *vsfObj = (VSF_Obj *)vsfHandle;
239 
240  vsfObj->pwmCounter++;
241 
242  // setup the Counter-Compare Control Register (CMPCTL)
243  if(vsfObj->state == VSF_STATE_PERIOD_SET)
244  {
245  if(vsfObj->pwmPeriod != vsfObj->pwmPeriodNow)
246  {
247  if(vsfObj->pwmPeriod > vsfObj->pwmPeriodNow)
248  {
249  vsfObj->pwmPeriodNow++;
250  }
251  else if(vsfObj->pwmPeriod < vsfObj->pwmPeriodNow)
252  {
253  vsfObj->pwmPeriodNow--;
254  }
255  }
256  else if(vsfObj->pwmCounter > vsfObj->pwmWaitTime)
257  {
258  vsfObj->pwmCounter = 0;
259  vsfObj->state = VSF_STATE_IDLE;
260  }
261  }
262 
263  return;
264 }
